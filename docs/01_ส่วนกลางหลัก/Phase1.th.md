bucket = "genesis_world"
url = "http://database:8086"

db_client = InfluxDBClient(url=url, token=token, org=org)
write_api = db_client.write_api(write_options=SYNCHRONOUS)

def save_to_db(sector, device_type, data):
    # ‡∏™‡∏£‡πâ‡∏≤‡∏á‡∏à‡∏∏‡∏î‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏• (Point) ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö‡πÄ‡∏Å‡πá‡∏ö‡πÉ‡∏ô Database
    point = Point("environment_metrics") \
        .tag("sector", sector) \
        .tag("device", device_type)
    
    # ‡∏ß‡∏ô‡∏•‡∏π‡∏õ‡πÄ‡∏Å‡πá‡∏ö‡∏Ñ‡πà‡∏≤‡∏ó‡∏∏‡∏Å‡∏ï‡∏±‡∏ß‡∏ó‡∏µ‡πà‡∏™‡πà‡∏á‡∏°‡∏≤ (‡πÄ‡∏ä‡πà‡∏ô temperature, humidity)
    for key, value in data.items():
        if isinstance(value, (int, float)):
            point.field(key, value)
    
    write_api.write(bucket=bucket, org=org, record=point)
    print(f"üíæ Data from {sector} saved to InfluxDB")

# ‡∏ô‡∏≥‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô‡∏ô‡∏µ‡πâ‡πÑ‡∏õ‡πÄ‡∏£‡∏µ‡∏¢‡∏Å‡πÉ‡∏ä‡πâ‡πÉ‡∏ô on_message ‡∏Ç‡∏≠‡∏á‡∏Ñ‡∏∏‡∏ì

#### **3. ‡πÑ‡∏ü‡∏•‡πå `main.py` (Core AI Engine)**

Code ‡∏ô‡∏µ‡πâ‡∏ñ‡∏π‡∏Å‡∏≠‡∏≠‡∏Å‡πÅ‡∏ö‡∏ö‡∏°‡∏≤‡πÉ‡∏´‡πâ‡∏£‡∏≠‡∏á‡∏£‡∏±‡∏ö‡πÇ‡∏Ñ‡∏£‡∏á‡∏™‡∏£‡πâ‡∏≤‡∏á‡πÅ‡∏ö‡∏ö Ecosystem ‡πÇ‡∏î‡∏¢‡∏Å‡∏≤‡∏£‡∏Ñ‡∏±‡∏î‡πÅ‡∏¢‡∏Å‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏• (Topic) ‡πÅ‡∏•‡∏∞‡∏°‡∏µ‡∏™‡πà‡∏ß‡∏ô‡∏Ç‡∏≠‡∏á **AI Rules** ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡∏™‡∏±‡πà‡∏á‡∏Å‡∏≤‡∏£‡∏Ç‡πâ‡∏≤‡∏°‡∏£‡∏∞‡∏ö‡∏ö‡∏Ñ‡∏£‡∏±‡∏ö

```python
import paho.mqtt.client as mqtt
import json

# --- ‡∏Å‡∏≤‡∏£‡∏ï‡∏±‡πâ‡∏á‡∏Ñ‡πà‡∏≤‡πÄ‡∏ö‡∏∑‡πâ‡∏≠‡∏á‡∏ï‡πâ‡∏ô ---
MQTT_BROKER = "mqtt"  # ‡∏ä‡∏∑‡πà‡∏≠ Service ‡πÉ‡∏ô Docker Compose
MQTT_PORT = 1883
# Subscribe ‡∏ó‡∏∏‡∏Å‡∏≠‡∏¢‡πà‡∏≤‡∏á‡∏ó‡∏µ‡πà‡∏Ç‡∏∂‡πâ‡∏ô‡∏ï‡πâ‡∏ô‡∏î‡πâ‡∏ß‡∏¢ genesis/
MQTT_TOPIC = "genesis/#" 

# --- ‡∏£‡∏∞‡∏ö‡∏ö AI Rules (Logic ‡∏Å‡∏≤‡∏£‡∏ï‡∏±‡∏î‡∏™‡∏¥‡∏ô‡πÉ‡∏à‡∏Ç‡πâ‡∏≤‡∏°‡∏£‡∏∞‡∏ö‡∏ö) ---
def ai_logic_engine(sector, device_type, data):
    print(f"üß† AI Analysing data from [{sector}] - {device_type}: {data}")
    
    # ‡∏ï‡∏±‡∏ß‡∏≠‡∏¢‡πà‡∏≤‡∏á Rule 1: ‡∏ñ‡πâ‡∏≤‡∏ü‡∏≤‡∏£‡πå‡∏°‡∏£‡πâ‡∏≠‡∏ô (Farm) ‡πÉ‡∏´‡πâ‡∏™‡∏±‡πà‡∏á‡∏ö‡πâ‡∏≤‡∏ô‡∏õ‡∏¥‡∏î‡∏°‡πà‡∏≤‡∏ô (Home)
    if sector == "farm" and device_type == "sensor":
        temp = data.get("temperature", 0)
        if temp > 35:
            print("‚ö†Ô∏è ALERT: Farm is too hot! Sending command to Smart Home...")
            client.publish("genesis/home/actuator/curtain", json.dumps({"action": "close"}))

    # ‡∏ï‡∏±‡∏ß‡∏≠‡∏¢‡πà‡∏≤‡∏á Rule 2: ‡∏ñ‡πâ‡∏≤‡∏°‡∏µ‡∏Ñ‡∏ô‡∏ö‡∏∏‡∏Å‡∏£‡∏∏‡∏Å‡∏ö‡πâ‡∏≤‡∏ô (Home) ‡πÉ‡∏´‡πâ‡∏™‡∏±‡πà‡∏á‡∏´‡∏∏‡πà‡∏ô‡∏¢‡∏ô‡∏ï‡πå‡πÑ‡∏õ‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö (Robot)
    if sector == "home" and data.get("alert") == "intruder":
        print("üö® ALERT: Intruder detected! Dispatching Robot...")
        client.publish("genesis/robot/command/move", json.dumps({"target": "living_room", "mode": "security"}))

# --- Callback ‡πÄ‡∏°‡∏∑‡πà‡∏≠‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏°‡∏ï‡πà‡∏≠‡∏™‡∏≥‡πÄ‡∏£‡πá‡∏à ---
def on_connect(client, userdata, flags, rc):
    print(f"‚úÖ Core Genesis connected with result code {rc}")
    client.subscribe(MQTT_TOPIC)

# --- Callback ‡πÄ‡∏°‡∏∑‡πà‡∏≠‡∏°‡∏µ‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏™‡πà‡∏á‡πÄ‡∏Ç‡πâ‡∏≤‡∏°‡∏≤ ---
def on_message(client, userdata, msg):
    try:
        # ‡πÅ‡∏õ‡∏•‡∏á‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏à‡∏≤‡∏Å JSON
        payload = json.loads(msg.payload.decode())
        
        # ‡πÅ‡∏¢‡∏Å‡πÅ‡∏¢‡∏∞‡∏ß‡πà‡∏≤‡∏°‡∏≤‡∏à‡∏≤‡∏Å Sector ‡πÑ‡∏´‡∏ô (genesis/home/sensor/...)
        topic_parts = msg.topic.split('/')
        if len(topic_parts) >= 3:
            sector = topic_parts[1]      # home, farm, or robot
            device_type = topic_parts[2] # sensor or actuator
            
            # ‡∏™‡πà‡∏á‡πÄ‡∏Ç‡πâ‡∏≤‡∏™‡∏°‡∏≠‡∏á AI
            ai_logic_engine(sector, device_type, payload)
            
    except Exception as e:
        print(f"‚ùå Error parsing message: {e}")

# --- ‡πÄ‡∏£‡∏¥‡πà‡∏°‡∏ï‡πâ‡∏ô‡∏Å‡∏≤‡∏£‡∏ó‡∏≥‡∏á‡∏≤‡∏ô ---
client = mqtt.Client()
client.on_connect = on_connect
client.on_message = on_message

print("üöÄ Core Genesis AI Engine is starting...")
client.connect(MQTT_BROKER, MQTT_PORT, 60)
client.loop_forever()

```

  static unsigned long lastMsg = 0;

    Serial.println("üåæ Sent Farm Data");

    char homeBuffer[256];
    serializeJson(homeDoc, homeBuffer);

    Serial.println("üè† Sent Home Data");

    // 3. ‡∏à‡∏≥‡∏•‡∏≠‡∏á‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏à‡∏≤‡∏Å Smart Robot
